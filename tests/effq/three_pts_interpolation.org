  import numpy as np
  import tred
  import importlib
  import tred.raster.steps_mod
  # importlib.reload(tred.raster.steps_mod)
  from tred.util import iter_tensor_chunks
  from tred.raster.steps_mod import eval_qeff, eval_q, add_patches
  from tred.raster.steps import eval_qeff as eval_qeff_orig

  add_patches()
  nbchunk = 10
  qdiff = []
  qdiff_unique = []
  for ichunk, idrifted in enumerate(
            iter_tensor_chunks([dsigma, drifted[1], dcharge, dtail, dhead,
                                pg_offset],
                               chunk_size=nbchunk)):
      idsigma, idtime, idcharge, idtail, idhead, ipg_offset = idrifted

      partitions = torch.tensor([10, 10, 1]).to(device)
      qeff, effoffset = eval_qeff(idcharge, idtail, idhead, idsigma,
                   ipg_offset * partitions,
                   pg_shape * partitions,
                   origin=(0, 0, 0),
                   grid_spacing=torch.tensor(pixel_grid_spacing).to(device) / partitions,
                   method='gauss_legendre', npoints=(4, 4, 4))
      qeff = qeff.cpu().numpy()
      effoffset = effoffset.cpu().numpy()
      assert ~np.any(np.isnan(qeff))

      qcharge, qoffset = eval_q(idcharge, idtail, idhead, idsigma,
                          ipg_offset * partitions,
                          pg_shape * partitions,
                          origin=(0, 0, 0),
                          grid_spacing=torch.tensor(pixel_grid_spacing).to(device) / partitions,
                          method='gauss_legendre', npoints=(2,2,2))
      qcharge = qcharge.cpu().numpy()
      qoffset = qoffset.cpu().numpy()
      assert ~np.any(np.isnan(qcharge))

      assert np.array_equal(qoffset, effoffset)

      # write a comparisokn
      ndiv = [qeff.shape[i]//partitions[i].item() for i in range(-3, 0, 1)]
      nparts = [partitions[i].item() for i in range(-3, 0, 1)]
      assert all(qeff.shape[i] % partitions[i].item() == 0 for i in range(-3, 0, 1))
      new_view = [-1]
      for i in range(3):
          new_view.append(nparts[i])
          new_view.append(ndiv[i])
      qeff = qeff.reshape(new_view)
      qcharge = qcharge.reshape(new_view)
      qeff = qeff.sum(axis=(1,3,5))
      qcharge = qcharge.sum(axis=(1,3,5))
      dq = qeff - qcharge
      qdiff.extend(dq.flatten().tolist())

      qoffset = torch.as_tensor(qoffset)
      effoffset = torch.as_tensor(effoffset)
      qeff = torch.as_tensor(qeff)
      qcharge = torch.as_tensor(qcharge)
      unique_effoffsets, inverse_indices_eff = torch.unique(effoffset,
                                                            return_inverse=True,
                                                            sorted=True,
                                                            dim=0)
      unique_qoffsets, inverse_indices_q = torch.unique(qoffset,
                                                        return_inverse=True,
                                                        sorted=True,
                                                        dim=0)
      assert torch.equal(unique_effoffsets, unique_qoffsets)
      q_unique = torch.zeros((unique_qoffsets.size(0), *qcharge.shape[1:]),
                             dtype=qeff.dtype)
      q_unique.index_add_(0, inverse_indices_q, qcharge)
      qeff_unique = torch.zeros((unique_effoffsets.size(0), *qeff.shape[1:]),
                                dtype=qeff.dtype)
      qeff_unique.index_add_(0, inverse_indices_eff, qeff)
      dq_unique = qeff_unique - q_unique
      qdiff_unique.extend(dq_unique.flatten().tolist())

  print('Finished')
