Let us introduce a new version of charge calculation. We drop the
support for trilinear interpolation from field responses.  All charges
here are assigned to the lower corner of each interval, instead of
using =convolution=.
#+begin_src jupyter-python :results no
  from tred.raster.steps import create_wu_block, create_node1ds, qline_diff3D, create_w_block
  def eval_q(Q, X0, X1, Sigma, offset, shape, origin, grid_spacing, method, npoints):
        '''
        Args:
            Q (Nsteps, )
            X0 (Nsteps, 3)
            X1 (Nsteps, 3)
            Sigma (Nsteps, 3)
            offset (Nsteps, vdim)
            shape (vdim,)
            origin (vdim,)
            grid_spacing (vdim,)
            npoints (vdim, )
        Return:
            effective charge

        FIXME:
           w block, u blocks supports vdim = any number,
           offset, shape, origin, they also support vdim = any number,
           Q, X0, X1, Sigma, their vdim is always 3.
        FIXME:
           vdim of npoints in principle can be different from offset, shape, origin, spacing.
           Now it is fixed to the same as others.
        '''
        if not isinstance(Q, torch.Tensor):
            raise ValueError('Q must be a torch.Tensor')
        device = Q.device

        kernel = create_w_block(method, npoints, grid_spacing, device)

        x, y, z = create_node1ds(method, npoints, origin, grid_spacing, offset, shape, device)

        charge = qline_diff3D(Q, X0, X1, Sigma,
                              x[:, :, None, None, :, None, None], # (Nsteps, L, 1, 1, I-1, 1, 1)
                              y[:, None, :, None, None, :, None], # (Nsteps, 1, M, 1, 1, J-1, 1)
                              z[:, None, None, :, None, None, :]) # (Nsteps, 1, 1, N, 1, 1, K-1)      print(charge.shape)
        charge = charge * kernel[None,:,:,:,None,None,None]

        charge = charge.sum(dim=(1,2,3))

        return charge

  def helper(method, *args, **kwds):
      return method(*args, **kwds)

  def too_small(X0, X1, Sigma, threshold=0.05):
      sqrt2 = 1.4142135623730951
      x0, y0, z0 = [X0[:, i] for i in range(3)]
      x1, y1, z1 = [X1[:, i] for i in range(3)]
      sx, sy, sz = [Sigma[:, i] for i in range(3)]

      # Calculate differences
      dx01 = x0 - x1
      dy01 = y0 - y1
      dz01 = z0 - z1

      # Calculate squared terms
      sxsy2 = (sx*sy)**2
      sxsz2 = (sx*sz)**2
      sysz2 = (sy*sz)**2
      sx2 = sx**2
      sy2 = sy**2
      sz2 = sz**2

      # Calculate delta terms
      deltaSquare = (
          sysz2 * dx01**2 +
          sxsy2 * dz01**2 +
          sxsz2 * dy01**2
      )
      deltaSquareSqrt = torch.sqrt(deltaSquare)

      return torch.all(deltaSquareSqrt[:,None] < Sigma * threshold, dim=1)

  def qpoint_diff3D(Q, X0, X1, Sigma, x, y, z):
      """
      Args:
          Q (N,)
          X0 (N, 3)
          X1 (N, 3)
          Sigma (N, 3)
          x (N, other shape)
          y (N, other shape)
          z (N, other shape)
      return:
          q (N, othter shape)
      """
      num_dims_to_add = x.ndim - 1
      shape_new = [-1] + num_dims_to_add * [1] # TorchScript does NOT support (-1,) + dynamic tuples

      # Prepare for broadcasting
      x0, y0, z0 = tuple(X0[:,i].view(shape_new) for i in range(3))
      x1, y1, z1 = tuple(X1[:,i].view(shape_new) for i in range(3))
      sx, sy, sz = tuple(Sigma[:,i].view(shape_new) for i in range(3))
      Q = Q.view(shape_new)
      args = (Q, x0, y0, z0, x1, y1, z1, sx, sy, sz, x, y, z)
      xc, yc, zc = (x0+x1)/2, (y0+y1)/2, (z0+z1)/2
      charge = Q / torch.pow(torch.tensor(2*3.1415926), 1.5) / sx / sy / sz \
          ,* torch.exp(- (x - xc)**2 / 2 / sx**2
                      - (y - yc)**2 / 2 / sy**2
                      - (z - zc)**2 / 2 / sz**2)
      return charge

  def tred_function(Q, X0, X1, Sigma,
                    pg_offset, pg_shape, pixel_grid_spacing, partitions,
                    npoints, **kwds):
      qeff, _ = ts.eval_qeff(Q, X0, X1, Sigma,
                             pg_offset * partitions,
                             pg_shape * partitions,
                             origin=(0, 0, 0),
                             grid_spacing=pixel_grid_spacing.to(device) / partitions,
                             method='gauss_legendre', npoints=npoints,
                             xyz_limit=torch.tensor((100000, 1000000, 100000)).to(device),
                             ,**kwds)
      qeff2, _ = ts.eval_qeff(Q, X0, X1, Sigma,
                             pg_offset * partitions,
                             pg_shape * partitions,
                             origin=(0, 0, 0),
                             grid_spacing=pixel_grid_spacing.to(device) / partitions,
                             method='gauss_legendre', npoints=npoints,
                             qmodel=qpoint_diff3D,
                             xyz_limit=torch.tensor((100000, 1000000, 100000)).to(device),
                             ,**kwds)
      point_like = too_small(X0, X1, Sigma, 0.05)
      qeff = torch.where(point_like[:, None, None, None],
                         qeff2, qeff)
      assert ~torch.any(torch.isnan(qeff))

      # reshape to pixel level
      qeff = torch.nn.functional.pad(qeff, pad=(
          0, partitions[-1] - qeff.shape[-1] % partitions[-1],
          0, partitions[-2] - qeff.shape[-2] % partitions[-2],
          0, partitions[-3] - qeff.shape[-3] % partitions[-3],
      ))
      qeff = qeff.view(-1, qeff.shape[-3] // partitions[-3], partitions[-3],
                qeff.shape[-2] // partitions[-2], partitions[-2],
                qeff.shape[-1] // partitions[-1], partitions[-1])
      qeff = qeff.sum(dim=(-5, -3, -1))
      return qeff

  def compute_diff(ten0, ten1, threshold=10):
      ret = torch.where(ten0 > threshold, (ten0-ten1)/ten0,
                  torch.where(ten1 > threshold, (ten0-ten1)/ten1, 0))
